Facebook\HHAPIDoc\DocBlock\ParameterInfo

The documentation information for a parameter

namespace Facebook\HHAPIDoc\DocBlock;

type ParameterInfo = shape(
  'name' => string,
  'types' => vec<string>,
  'text' => ?string,
);

Fields

  • 'name': string - The name of the parameter
  • 'types': vec<string> - The valid types for the parameter
  • 'text': ?string - The human-readable documentation for the parameter
